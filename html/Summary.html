number-≅-Σ : ∀{p} → Number p ≅ NumberInterpretation p
number-≅-Σ = λ where
  .Iso.fun               → pop
  .Iso.inv      (x ,  p) → x ,, p
  .Iso.leftInv  (x ,, p) → refl
  .Iso.rightInv (x ,  p) → refl
